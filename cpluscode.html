Calculator Program in C++

#include <iostream>
#include <cmath>

using namespace std;

// Function declarations
void showMenu();
double add(double a, double b);
double subtract(double a, double b);
double multiply(double a, double b);
double divide(double a, double b);
double power(double base, double exponent);
double sqrt(double a);

int main() {
  double num1, num2;
  int choice;
  bool continueCalculation = true;

  while (continueCalculation) {
    showMenu();
    cout << "Enter your choice: ";
    cin >> choice;

    switch (choice) {
      case 1:
        cout << "Enter two numbers: ";
        cin >> num1 >> num2;
        cout << "Result: " << add(num1, num2) << endl;
        break;
      case 2:
        cout << "Enter two numbers: ";
        cin >> num1 >> num2;
        cout << "Result: " << subtract(num1, num2) << endl;
        break;
      case 3:
        cout << "Enter two numbers: ";
        cin >> num1 >> num2;
        cout << "Result: " << multiply(num1, num2) << endl;
        break;
      case 4:
        cout << "Enter two numbers: ";
        cin >> num1 >> num2;
        if (num2 != 0) {
          cout << "Result: " << divide(num1, num2) << endl;
        } else {
          cout << "Error: Division by zero!" << endl;
        }
        break;
      case 5:
        cout << "Enter the base and exponent: ";
        cin >> num1 >> num2;
        cout << "Result: " << power(num1, num2) << endl;
        break;
      case 6:
        cout << "Enter a number: ";
        cin >> num1;
        if (num1 >= 0) {
          cout << "Result: " << sqrt(num1) << endl;
        } else {
          cout << "Error: Negative number!" << endl;
        }
        break;
      case 7:
        continueCalculation = false;
        break;
      default:
        cout << "Invalid choice. Please try again." << endl;
        break;
    }
  }

  cout << "Calculator has exited." << endl;
  return 0;
}

void showMenu() {
  cout << "\n***** Calculator Menu *****\n";
  cout << "1. Addition\n";
  cout << "2. Subtraction\n";
  cout << "3. Multiplication\n";
  cout << "4. Division\n";
  cout << "5. Power\n";
  cout << "6. Square Root\n";
  cout << "7. Exit\n";
  cout << "****************************\n";
}

double add(double a, double b) {
  return a + b;
}
double subtract(double a, double b) {
  return a - b;
}
double multiply(double a, double b) {
  return a * b;
}
double divide(double a, double b) {
  return a / b;
}
double power(double base, double exponent) {
  return pow(base, exponent);
}
double sqrt(double a) {
  return std::sqrt(a);
}

This program is a calculator written in C++ that replicates the functions of an Android mobile calculator. The program includes addition, subtraction, multiplication, division, power, and square root functions.

1. showMenu function: This function displays the calculator's menu, allowing the user to select different operations.

2. Addition function: The add(double a, double b) function adds two numbers and returns the result.

3. Subtraction function: The subtract(double a, double b) function subtracts two numbers and returns the result.

4. Multiplication function: The multiply(double a, double b) function multiplies two numbers and returns the result.

5. Division function: The divide(double a, double b) function divides two numbers and returns the result. It handles division by zero errors.

6. Power function: The power(double base, double exponent) function calculates the power of a base number raised to an exponent and returns the result.

7. Square Root function: The sqrt(double a) function calculates the square root of a number. It handles errors for negative numbers.

8. Main function: The main function takes user input and calls the respective function. This loop continues until the user selects the exit option.
